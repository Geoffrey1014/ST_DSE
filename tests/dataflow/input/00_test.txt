PROGRAM Main
VAR
	IS_LEAD_EDGE: BOOL;
	PREV_LOAD: BOOL;
END_VAR

VAR_OUTPUT
	OUT_1: BOOL;
END_VAR

VAR_INPUT
	TLOAD: BOOL;
	RESET: BOOL;
	LOAD: BOOL;
	IN_1: BOOL;
END_VAR

IS_LEAD_EDGE := (NOT PREV_LOAD) AND TLOAD;
IF RESET THEN
    OUT_1 := TRUE;
ELSIF LOAD OR (IS_LEAD_EDGE) THEN
    OUT_1 := IN_1;

ELSIF IN_1 THEN
    OUT_1 := FALSE;
(*
ELSE
    LOAD := FALSE;
    *)
END_IF;
PREV_LOAD := TLOAD;

END_PROGRAM
