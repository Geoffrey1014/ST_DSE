PROGRAM Main
VAR
	IS_LEAD_EDGE: BOOL;
	PREV_LOAD: BOOL;
END_VAR

VAR_OUTPUT
	OUT_1: BOOL;
END_VAR

VAR_INPUT
	TLOAD: BOOL;
	RESET: BOOL;
	LOAD: BOOL;
	IN_1: BOOL;
	IN_2 : INT := 123.5;
END_VAR

    IS_LEAD_EDGE := (NOT PREV_LOAD) AND TLOAD;
    IF RESET THEN
        OUT_1 := TRUE;
    ELSIF LOAD OR (IS_LEAD_EDGE) THEN
        OUT_1 := IN_1;
    ELSE
        LOAD := FALSE;
    END_IF;
    PREV_LOAD := TLOAD;
    IN_2 := IN_1;
    IN_3 := IN_1 + IN_2;

END_PROGRAM
