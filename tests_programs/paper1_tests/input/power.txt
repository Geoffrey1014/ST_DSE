PROGRAM power
  VAR_INPUT
    X : INT ;
    Y : INT ;
  END_VAR
  VAR_OUTPUT
    RES : REAL := 1;
    FLAG : BOOL := TRUE;
  END_VAR
  VAR
    EXP :INT;
  END_VAR

IF Y > 0 THEN
    EXP := Y;
ELSE
    EXP := -Y;
END_IF;

WHILE EXP <> 0 DO
	RES := RES * X;
	EXP := EXP -1;
END_WHILE;

IF Y <= 0 THEN
    IF X = 0 THEN
        FLAG := FALSE;
    END_IF;
ELSE
    RES := 1 / RES;
END_IF;

END_PROGRAM
